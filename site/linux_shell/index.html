# Pseudo Code
a = 1;
b = "string";
c = [1, 2, "str", true, false];
a = 1 + 2
b = a * 7
c = "Con" ++ "cat";
d = c ++ b

# Bash
a=$((1))
b="string"
c=($((1)) $((2)) "str" $((1)) $((0)))
a=$((1 + 2))
b=$(($a * 7))
c="Con""cat"
d="$c""$b"

Condition

# Pseudo Code
a = 3;
if (a > 2) {
  println("Yes");
} else {
  println("No");
}

# else if
a = 10;
if (a > 7) {
  println("Big");
} else if (a > 4) {
  println("Medium");
} else {
  println("Small");
}

# Bash
a=$((3))
if [ $(($a > 2)) == 1 ]; then
  "echo" "-e" "Yes"
else
  "echo" "-e" "No"
fi

# elif
a=5
if [ "$a" -gt 7 ]; then
  echo "Big"
elif [ "$a" -gt 4 ]; then
  echo "Medium"
else
  echo "Small"
fi

Loop

# Pseudo Code: Fibonacci
n = 0;
i = 0;
j = 1;
while (n < 60) {
  k = i + j;
  i = j;
  j = k;
  n = n + 1;
  echo(k);
}

# Bash: Fibonacci
n=0
i=0
j=1
while [ "$n" -lt 60 ]; do
  k=$(($i + $j))
  i="$j"
  j="$k"
  n=$(($n + 1))
  "echo" "$k"
done
